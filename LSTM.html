import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
from sklearn.preprocessing import MinMaxScaler
from tensorflow.keras.models import Sequential
from tensorflow.keras.layers import LSTM, Dense

# Read file
try:
    df = pd.read_excel(r"C:\Users\Dell\Downloads\stock_data.xls", engine='xlrd')
except:
    df = pd.read_csv(r"C:\Users\Dell\Downloads\stock_data.xls")

# Show column names
print("Available columns:", df.columns)

# Convert all column names to string for safety
df.columns = df.columns.astype(str)

# Try to auto-detect 'Close' column
close_col = None
for col in df.columns:
    if 'close' in col.lower():
        close_col = col
        break

if close_col is None:
    raise ValueError("No column found that looks like a 'Close' price.")

# Drop rows where 'Close' is NaN or not a number
df = df[pd.to_numeric(df[close_col], errors='coerce').notnull()]
df[close_col] = df[close_col].astype(float)

# Get only Close column
data = df[[close_col]].values

# Normalize
scaler = MinMaxScaler()
scaled_data = scaler.fit_transform(data)

# Create sequences
X, y = [], []
for i in range(60, len(scaled_data)):
    X.append(scaled_data[i - 60:i])
    y.append(scaled_data[i])

X, y = np.array(X), np.array(y)
X = X.reshape((X.shape[0], X.shape[1], 1))

# Build LSTM model
model = Sequential()
model.add(LSTM(50, return_sequences=True, input_shape=(X.shape[1], 1)))
model.add(LSTM(50))
model.add(Dense(1))
model.compile(optimizer='adam', loss='mean_squared_error')
model.fit(X, y, epochs=5, batch_size=32)

# Forecast next 30 values
input_seq = scaled_data[-60:].reshape(1, 60, 1)
predicted = []

for _ in range(30):
    next_val = model.predict(input_seq, verbose=0)[0][0]
    predicted.append(next_val)
    input_seq = np.append(input_seq[:, 1:, :], [[[next_val]]], axis=1)

# Inverse transform
predicted_prices = scaler.inverse_transform(np.array(predicted).reshape(-1, 1))

# Plot
plt.figure(figsize=(12, 6))
plt.plot(df[close_col].values, label="Historical")

# If date column exists, use it
if 'Date' in df.columns or 'date' in [col.lower() for col in df.columns]:
    date_col = next(col for col in df.columns if col.lower() == 'date')
    future_dates = pd.date_range(df[date_col].iloc[-1], periods=31, freq="B")[1:]
    plt.plot(future_dates, predicted_prices, label="LSTM Forecast", color="orange")
else:
    plt.plot(range(len(df), len(df) + 30), predicted_prices, label="LSTM Forecast", color="orange")

plt.title("LSTM Forecast (Next 30 Days)")
plt.xlabel("Date")
plt.ylabel("Price")
plt.legend()
plt.grid()
plt.show()
